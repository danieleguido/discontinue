Discontinue.keepon

Execute an ajax request every 2500 milliseconds (since last completed ajax request): 
		$(window).ready(function(){
			// listen to all discontinue events
			$.fn.discontinue.verbose( function( event, data ){
				$("#console").append( [ "\n", "[", event.namespace,"]", new Date(), event.type, data].join(" ") )
			});
			
			// listen to success event
			$.fn.discontinue.on( 'success.keepon.discontinue', function(event, data){
				console.log(data);
			});

			// execute standard keepon
			$().discontinue.keepon({
				delay: 1200,
				url:"response.json"
			});
		});
